202void list_Class::drawOptions(){
203 //Create text sprite in RAM & Set text options
204 M5Canvas text(&Display);
205 text.createSprite(this->size.x -2*this->border.x, this->size.y -3*this->border.y -rowSize(this->titleSize));
206 text.setTextColor(this->textColour);
207 text.setTextSize(this->textSize);
208 text.setCursor(0,1);
209 //For loop to draw the options (taking current pos into account)
210 int index = pos -2;
211 if (index < 0) index = 0; //If the selected pos is on 0,1,2 drawing starts @ index 0
212 else if (this->elementNum - index -1 < this->rows) index = this->elementNum-this->rows; //If the selected position is near the end, print enough to fill the rows
213 int count = 0;
214 for(int i = index; (i <this->elementNum && count < this->rows); i++){
215 if (i == pos) {
216 text.setColor(this->highlightColour);
217 int rows = list_Class::rowsOccupied(this->elementName[i].length() +1, this->textSize, this->size.x -2*this->border.x);
218 text.fillRoundRect(text.getCursorX()-1, text.getCursorY()-2, size.x -2*this->border.x, rows*rowSize(this->textSize)+1, 5);
219 text.setTextColor(this->backColour);
220 text.println(" "+this->elementName[i]);
221 text.setTextColor(this->textColour);
222 } else text.println(this->elementName[i]);
223 count++;
224 }
225
226 text.pushSprite(origin.x +this->border.x, origin.y +2*this->border.y + rowSize(this->titleSize));
227 text.deleteSprite();
