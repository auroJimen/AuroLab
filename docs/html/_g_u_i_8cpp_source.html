170void textBox_Class::enableTextInput(){
171
172 //First we change mode to text input, and increase the size of the buffer if it's not already set that way
173 if(Buffer.Mode != mode::text || Buffer.getBufferSize() != DEF_BUFFER_SIZE_TEXT_MODE
174 || eTaskGetState(Buffer.getTaskHandle()) == eTaskState::eDeleted
175 || eTaskGetState(Buffer.getTaskHandle()) == eTaskState::eInvalid) {
176 //Checks wether the Buffer is set to an incorrect mode, or the keyboard task has been killed
177 //if either of these conditions are met, it reinstantiates the buffer in the correct mode
178 Buffer.killTask(); //Kill current buffer class, free up memory
179 Buffer = buffer_Class(mode::text); //Reinitilize Buffer in text mode w/DEF_BUFFER_SIZE_TEXT_MODE buffer
180 Buffer.begin(); //Create keyboard task and begin running it

191void textBox_Class::update(){
192 //Updates the text part of the sprite with the curent contents of the buffer
193 M5Canvas text = (this->parent == nullptr) ? M5Canvas(&Display): M5Canvas(parent);
194 text.createSprite(this->size.x -2*this->border.x, this->size.y -2*this->border.y);
195 text.setTextWrap(true);
196 text.setTextScroll(true);
197 text.setTextSize(this->textSize);
198 text.setTextColor(this->textColour);
199
200 String aux = Buffer.getDataStr();
201 if (Buffer.cursor < this->dispLen) { //Text from 0 to cursor fits
202 //Print everything before the cursor
203 text.print(aux.substring(0, Buffer.cursor));
204
205 //Print cursor
206 text.setColor(this->cursorColour);
207 text.setTextColor(this->backColour);
208 text.fillRect(text.getCursorX()-1, text.getCursorY()-1, this->cursorSize.x, this->cursorSize.y);
209 text.print(aux[Buffer.cursor]);
210 text.setTextColor(this->textColour);
211
212 //Print everything after the cursor that still fits
213 int end = (aux.length() > this->dispLen) ? this->dispLen : aux.length();
214 text.print(aux.substring(Buffer.cursor+1, end));
215 }
216 else { //Text from 0 to cursor doesn't fit
217 //Shift the start to the right as much as the cursor is greater than the display length
218 text.print(aux.substring(Buffer.cursor - (this->dispLen -1), Buffer.cursor));
219
220 //Print cursor
221 text.setColor(this->cursorColour);
222 text.setTextColor(this->backColour);
223 text.fillRect(text.getCursorX()-1, text.getCursorY()-1, this->cursorSize.x, this->cursorSize.y);
224 text.print(aux[Buffer.cursor]);
225 text.setTextColor(this->textColour);
226
227 //There's no space to print anything after
228
229 }
230 //Push & delete
231 text.pushSprite(this->pos.x + this->border.x, this->pos.y + this->border.y);
232 text.deleteSprite();

417void wifiMenu_Class::appLoop(){
418 //Perform primary scan
419 if(this->elementNum== 0)this->scan();
420 //Draw the menu
421 this->draw();
422 bool flag = false;
423 //Check keyboard mode & that it is active
424 while(!flag){
425 //Endless application loop
426 //Nav signal listener
427 vTaskDelay(10 / portTICK_PERIOD_MS); //Alows other things to update, housekeeping etc
428 if (Buffer.signal != navSignal::NP) {
429 log_i("Nav signal detected");
430 Buffer.keyboardEnable = false; //Deactivate keyboard during execution
431 switch (Buffer.signal){
432 case navSignal::ESC: //Close the menu, UNIMPLEMENTED
433 log_i("Close event");
434 break;
435 case navSignal::UP:
436 case navSignal::LEFT: //Scroll up
437 log_i("Scroll up event");
438 this->scrollUp();
439 break;
440 case navSignal::DOWN:
441 case navSignal::RIGHT: //Scroll down
442 log_i("Scroll down event");
443 this->scrollDown();
444 break;
445 case navSignal::ENTER: //Element selected event
446 log_i("Element selected event");
447 this->enterEvent();
448 //Prepare to exit the loop and return
449 flag = true;
450 this->del();
451 break;
452 case navSignal::OPT: //Option event
453 log_i("Option event");
454 this->optnEvent();
455 //After the opt event it return to the appLoop
456 this->draw();
457 break;
458 case navSignal::CTRL: //Rescan event
459 log_i("Rescan event");
460 this->scan();
461 this->draw();
462 }
463 Buffer.signal = navSignal::NP; //Clear signal flag
464 Buffer.keyboardEnable = true; //Reactivate after
465 log_i("Keyboard input reactivated");
466 }
467 }

654void topBarLoop(void* parameter){
655 log_i("topBar updater task created");
656 //Create the interrupt we'll use to time when we update timed things on screen
657 hw_timer_t* timer = timerBegin(3, 80, true);
658 timerAttachInterrupt(timer, &timerISR, false); //Atach interrupt to the timer
659 timerAlarmWrite(timer, 1000000, true); //Set the alarm that triggers the interrupt
660 timerAlarmEnable(timer); //Enable the alarm
661 //Suspend the current task (it'll resume from here once the interrupt is triggered)
662 vTaskSuspend(NULL);
663 //Infinite loop, each time xTaskResumeFromISR is called from within the ISR, a round of this loop will
664 //be executed, succesive calls to resuming won't affect an already resumed task, & the loop suspends the task
