84void keyBoardLoop(void* parameters) {
85 log_i("Keyboard task created");
86 //Create the interrupt we'll use to time how many times per second we check the keyboard input
87 timerAttachInterrupt(Buffer.getHardwareTimer(), &keyBoardISR, false); //Atach interrupt to the timer
88 timerAlarmWrite(Buffer.getHardwareTimer(), 50000, true); //Set the alarm that triggers the interrupt
89 timerAlarmEnable(Buffer.getHardwareTimer()); //Enable the alarm
90 //Suspend the current task (it'll resume from here once the interrupt is triggered)
91 vTaskSuspend(NULL);
92 //Infinite loop, each time xTaskResumeFromISR is called from within the ISR, a round of this loop will
93 //be executed, succesive calls to resuming won't affect an already resumed task, & the loop suspends the task
94 //once the round is finished
95 for(;;){
96 Keyboard_Class::KeysState status = M5Cardputer.Keyboard.keysState();
97 //Handle keyboard presses
98 if (Buffer.Mode == mode::text) {
99 //We are in text entry mode
100
101 if (status.word.size() == 0) {
102 //Special cases
103 if (status.del) {
104 if (status.fn) {
105 //supr / del
106 log_i("Supr pressed");
107 for(int i = Buffer.cursor+1; i < Buffer.getBufferSize() -1; i++){
108 if(Buffer.getData()[i] == '\0') break; //End of string reached
109 Buffer.getData()[i] = Buffer.getData()[i+1];
110 }
111 } else {
112 //Backspace
113 log_i("Backspace pressed");
114 if (Buffer.cursor != 0) {
115 Buffer.cursor --;
116 //If the next char is the null term, set to null term
117 if (Buffer.getData()[Buffer.cursor+1] == '\0') Buffer.getData()[Buffer.cursor] = '\0';
118 else {
119 //We are not @ the end of the string, must move back everything
120 for (int i = Buffer.cursor; i < Buffer.getBufferSize() -1; i++){
121 if(Buffer.getData()[i] == '\0') break; //End of string reached
122 else Buffer.getData()[i] = Buffer.getData()[i+1];
123 }
124 }
125 }
126
127 }
128 } else if (status.tab){
129 //Tab single press, insert \t
130 log_i("Tab pressed");
131 Buffer.getData()[Buffer.cursor] = '\t';
132 Buffer.cursor++;
133 } else if (status.enter){
134 if (status.fn) Buffer.signal = navSignal::ENTER;
135 else {
136 //Enter single press, insert \n
137 log_i("Enter pressed");
138 Buffer.getData()[Buffer.cursor] = '\n';
139 Buffer.cursor++;
140 }
141 } else if (status.alt || status.ctrl || status.fn || status.opt || status.shift || status.space) Buffer.signal = navSignal::NP;
142 else log_i("Keyboard release");
143
144 //Maybe implement opt, alt & ctrl? for text mode?
145 //Opt should bring out the conf menu for the programme,
146 //Alt & ctrl should be used in some way for keyboard shortcuts, but they should be programme specific,
147 //I'm not sure how i wanna implement that
148 //(maybe add a possible navSwicht style func given when creating the buffer?)
149 //Other possible idea is fn+shift = BLOCK MAYUS,
150 //only if i find an elegant way to show it is active to the user
151 } else {
152 log_i("Keyboard pressed, text entry mode, word is %i char", status.word.size());
153 for(int i= 0; i< status.word.size(); i++){
154 if (Buffer.cursor == Buffer.getBufferSize()) Buffer.clearBuffer(); //Check if the buffer is full
155 if (status.fn) {
156 //Nav controls for text editing
157 char aux = status.word[i];
158 //Export nav singals that should be handled externally
159 Buffer.signal = navSwitch(aux);
160 //Handle the internal ones (cursor movement)
161 //Cursor should never be negative
162 if (Buffer.signal == navSignal::LEFT && Buffer.cursor > 0) Buffer.cursor --;
163 //Cursor can be up to the next character to the end of the current string (to be able to enlarge it)
164 else if (Buffer.signal == navSignal::RIGHT && Buffer.cursor < Buffer.getDataStr().length()) Buffer.cursor ++;
165 } else if (status.opt){
166 //Option key pressed overwrite mode
167 Buffer.getData()[Buffer.cursor] = status.word[i];
168 Buffer.cursor++;
169 } else {
170 //Normal mode (insert text)
171 String aux = Buffer.getDataStr().substring(Buffer.cursor); //Save everythin thats after the cursor
172 Buffer.getData()[Buffer.cursor] = status.word[i];
173 Buffer.cursor++;
174 for(int i= 0; i < aux.length(); i++){
175 Buffer.getData()[Buffer.cursor +i] = aux[i];
176 }
177 }
178 }
179 }
180 if (CORE_DEBUG_LEVEL >= 5){
181 log_i("Buffer: \"%s\"", Buffer.getDataStr().c_str());
182 String aux = "";
183 int dataLen = Buffer.getDataStr().length();
184 for(int i = 0; i < dataLen +1; i++) aux+= (i == Buffer.cursor) ? '^' :' ';
185 log_i("Cursor: \"%s\"\n", aux);
186 log_i("Nav signal: %i\n", Buffer.signal);
187 }
188 } else {
189 //we are in navigation mode
190 if (status.word.size() ==1) {
191 log_i("Keyboard pressed, menu navigation mode");
192 char aux = status.word[0];
193 Buffer.signal = navSwitch(aux);
194 } else if (status.enter) {
195 Buffer.signal = navSignal::ENTER;
196 log_i("ENTER signal");
197 } else if(status.ctrl){
198 log_i("CTRL signal");
199 Buffer.signal = navSignal::CTRL;
200 } else if (status.opt){
201 Buffer.signal = navSignal::OPT;
202 log_i("OPT signal");
203 } else if (status.alt||status.del||status.fn||
204 status.shift||status.space) Buffer.signal = navSignal::NP;
205 else log_i("Keboard release");
206 log_i("Nav signal: %i\n", Buffer.signal);
207 }
208 //Suspend itself until it's called again
209 vTaskSuspend(NULL);
210 }
211}

213navSignal navSwitch(char aux){
214 switch(aux){
215 case '`':
216 case '~': log_i("Esc signal");
217 return navSignal::ESC;
218 case ';':
219 case ':': log_i("UP signal");
220 return navSignal::UP;
221 case '.':
222 case '>': log_i("DOWN signal");
223 return navSignal::DOWN;
224 case ',':
225 case '<': log_i("LEFT signal");
226 return navSignal::LEFT;
227 case '/':
228 case '?': log_i("RIGHT signal");
229 return navSignal::RIGHT;
230 default: return navSignal::NP;
231 }
